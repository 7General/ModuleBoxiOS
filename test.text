class GZCAppConfigManager : NSObject {

    - (NSInteger)newTerms {
        NSString * shareKey = @"GZCSharePersonInfoKey";
        NSUserDefaults.standardUserDefaults().setBool:forKey:(YES, shareKey);
        NSUserDefaults.standardUserDefaults().synchronize();
        return self.ORGnewTerms;
    }

}
